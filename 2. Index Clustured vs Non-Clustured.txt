 CREATE TABLE [Zyrafa](
	[Id] [int],
	[nazwa] [varchar](20),
	[wiek] [int]
)
----------------------------------------------------------------------------
Create PROCEDURE [dbo].[WstawDoZyrafy]
AS 
SET NOCOUNT ON
BEGIN
DECLARE @Id INT = 1
DECLARE @zyrafaNazwa VARCHAR(20)='zyrafa'
DECLARE @nazwa VARCHAR(20) = ''
DECLARE @wiek INT = 0

      WHILE @Id<200000
      BEGIN
      SET @nazwa=@zyrafaNazwa+CONVERT(varchar(20), @Id)
      SET @wiek=@Id%99 + 1

      INSERT INTO Zyrafa VALUES (@Id, @nazwa, @wiek)

      SET @Id+=1
      END
END

----------------------------------------------------------------------------
exec WstawDoZyrafy
----------------------------------------------------------------------------
drop table Zyrafa
delete from Zyrafa
